(function () { "use strict"; // Prevent XSS Attacks by sanitizing user inputs function sanitizeInput(input) { const tempDiv = document.createElement("div"); tempDiv.textContent = input; return tempDiv.innerHTML; } // Example: Sanitize all input fields automatically const inputs = document.querySelectorAll("input, textarea"); inputs.forEach(input => { input.addEventListener("input", (e) => { e.target.value = sanitizeInput(e.target.value); }); }); // Disable right-click to prevent unauthorized access document.addEventListener("contextmenu", (e) => { e.preventDefault(); console.warn("Right-click is disabled to enhance security."); }); // Prevent debugging tools (like F12/DevTools) document.addEventListener("keydown", (e) => { if (e.key === "F12" || (e.ctrlKey && e.shiftKey && e.key === "I")) { e.preventDefault(); console.warn("Developer tools are disabled."); } }); // Monitor DOM changes for suspicious activity const observer = new MutationObserver((mutations) => { mutations.forEach((mutation) => { if (mutation.addedNodes.length > 0) { mutation.addedNodes.forEach(node => { if (node.nodeType === 1 && node.tagName === "SCRIPT") { console.error("Unauthorized script detected and removed!"); node.remove(); } }); } }); }); observer.observe(document.body, { childList: true, subtree: true }); // Secure cookies document.cookie = "SecureCookie=SecureValue; Secure; HttpOnly; SameSite=Strict"; console.log("Basic security measures are active on this website."); })();
